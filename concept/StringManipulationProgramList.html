import java.util.*;

public class IntermediateSolutions {
    public static void main(String[] args) {
        // --- Reversal & Rotation [cite: 40-45] ---
        System.out.println(reverseSB("Java"));
        System.out.println(reverseManual("Java"));
        System.out.println(reverseOrderWords("Hello World Java"));
        System.out.println(reverseEachWord("Hello World"));
        System.out.println(isRotation("ABCD", "CDAB"));
        swapStrings("Hello", "World");

        // --- Palindromes & Anagrams [cite: 47-50] ---
        System.out.println(isPalindrome("Madam"));
        System.out.println(isAnagram("Keep", "Peek"));
        System.out.println(isPangram("The quick brown fox jumps over the lazy dog"));

        // --- Counting & Frequency [cite: 52-58] ---
        countDuplicates("programming");
        firstNonRepeating("swiss");
        countTypes("Java@123");

        // --- Modification & Regex [cite: 66-79] ---
        System.out.println(removeDupChars("banana"));
        System.out.println(toggleCase("HeLLo"));
        System.out.println(regexEmail("test@mail.com"));
    }

    // 40. Reverse SB
    static String reverseSB(String s) { return new StringBuilder(s).reverse().toString(); }

    // 41. Reverse Manual
    static String reverseManual(String s) {
        String r = "";
        for(int i=s.length()-1; i>=0; i--) r += s.charAt(i);
        return r;
    }

    // 42. Reverse Order of Words
    static String reverseOrderWords(String s) {
        String[] words = s.split(" ");
        StringBuilder sb = new StringBuilder();
        for(int i=words.length-1; i>=0; i--) sb.append(words[i]).append(" ");
        return sb.toString().trim();
    }

    // 43. Reverse Each Word
    static String reverseEachWord(String s) {
        String[] words = s.split(" ");
        StringBuilder sb = new StringBuilder();
        for(String w : words) sb.append(new StringBuilder(w).reverse()).append(" ");
        return sb.toString().trim();
    }

    // 44. Rotation Check
    static boolean isRotation(String s1, String s2) {
        return (s1.length() == s2.length()) && (s1+s1).contains(s2);
    }

    // 45. Swap without temp
    static void swapStrings(String a, String b) {
        a = a + b;
        b = a.substring(0, a.length() - b.length());
        a = a.substring(b.length());
        System.out.println("Swapped: " + a + " " + b);
    }

    // 47. Palindrome
    static boolean isPalindrome(String s) {
        String clean = s.replaceAll("\\s+", "").toLowerCase();
        return clean.equals(new StringBuilder(clean).reverse().toString());
    }

    // 48. Anagram
    static boolean isAnagram(String s1, String s2) {
        char[] a = s1.toLowerCase().toCharArray();
        char[] b = s2.toLowerCase().toCharArray();
        Arrays.sort(a); Arrays.sort(b);
        return Arrays.equals(a, b);
    }

    // 49. Pangram
    static boolean isPangram(String s) {
        boolean[] mark = new boolean[26];
        for(char c : s.toLowerCase().toCharArray()) 
            if(Character.isLetter(c)) mark[c - 'a'] = true;
        for(boolean b : mark) if(!b) return false;
        return true;
    }

    // 52. Count Duplicates
    static void countDuplicates(String s) {
        Map<Character, Integer> map = new HashMap<>();
        for(char c : s.toCharArray()) map.put(c, map.getOrDefault(c, 0) + 1);
        map.forEach((k,v) -> { if(v > 1) System.out.println(k + ":" + v); });
    }

    // 56. First Non-Repeating
    static void firstNonRepeating(String s) {
        Map<Character, Integer> map = new LinkedHashMap<>();
        for(char c : s.toCharArray()) map.put(c, map.getOrDefault(c, 0) + 1);
        for(Map.Entry<Character,Integer> e : map.entrySet()) {
            if(e.getValue() == 1) { System.out.println("First Unique: " + e.getKey()); return; }
        }
    }

    // 58. Count Types
    static void countTypes(String s) {
        int num=0, alpha=0, special=0;
        for(char c : s.toCharArray()) {
            if(Character.isDigit(c)) num++;
            else if(Character.isLetter(c)) alpha++;
            else special++;
        }
        System.out.println("Alpha: "+alpha+" Num: "+num+" Special: "+special);
    }

    // 66. Remove Duplicates
    static String removeDupChars(String s) {
        StringBuilder sb = new StringBuilder();
        s.chars().distinct().forEach(c -> sb.append((char)c));
        return sb.toString();
    }

    // 70. Toggle Case
    static String toggleCase(String s) {
        StringBuilder sb = new StringBuilder();
        for(char c : s.toCharArray()) {
            if(Character.isUpperCase(c)) sb.append(Character.toLowerCase(c));
            else sb.append(Character.toUpperCase(c));
        }
        return sb.toString();
    }

    // 75. Regex Email
    static boolean regexEmail(String e) { return e.matches("^[\\w-\\.]+@([\\w-]+\\.)+[\\w-]{2,4}$"); }
}

public class AdvancedSolutions {
    public static void main(String[] args) {
        // --- Substrings & Permutations [cite: 83-93] ---

        System.out.println(longestUniqueSub("abcabcbb"));
        printPermutations("ABC", "");
        printSubsequences("ABC", "");

        // --- Algorithms & Data [cite: 95-101] ---
        System.out.println("KMP Found: " + KMPSearch("ABABDABACDABABCABAB", "ABABCABAB"));
        System.out.println("Roman To Int: " + romanToInt("MCMXCIV"));
        System.out.println("Atoi: " + myAtoi("   -42"));
        System.out.println("Compress: " + compress("aaabbc"));

    // 83. Longest Palindromic Substring (Expansion)
    static String longestPalindrome(String s) {
        if(s==null || s.length()<1) return "";
        int start=0, end=0;
        for(int i=0; i<s.length(); i++) {
            int len1 = expand(s, i, i);
            int len2 = expand(s, i, i+1);
            int len = Math.max(len1, len2);
            if(len > end-start) { start = i-(len-1)/2; end = i+len/2; }
        }
        return s.substring(start, end+1);
    }
    static int expand(String s, int L, int R) {
        while(L>=0 && R<s.length() && s.charAt(L)==s.charAt(R)) { L--; R++; }
        return R-L-1;
    }

    // 84. Longest Substring No Repeats (Sliding Window)
    static int longestUniqueSub(String s) {
        Set<Character> set = new HashSet<>();
        int max=0, i=0, j=0;
        while(i<s.length() && j<s.length()) {
            if(!set.contains(s.charAt(j))) { set.add(s.charAt(j++)); max = Math.max(max, j-i); }
            else set.remove(s.charAt(i++));
        }
        return max;
    }

    // 86. Subsequences (Recursion)
    static void printSubsequences(String str, String ans) {
        if(str.length()==0) { System.out.print(ans+" "); return; }
        printSubsequences(str.substring(1), ans + str.charAt(0));
        printSubsequences(str.substring(1), ans);
    }

    // 91. Permutations
    static void printPermutations(String str, String ans) {
        if(str.length()==0) { System.out.print(ans+" "); return; }
        for(int i=0; i<str.length(); i++) {
            char ch = str.charAt(i);
            String ros = str.substring(0,i) + str.substring(i+1);
            printPermutations(ros, ans+ch);
        }
    }

    // 95. KMP Algorithm Pattern Search
    static boolean KMPSearch(String txt, String pat) {
        int M = pat.length(), N = txt.length();
        int[] lps = new int[M];
        int j = 0; 
        computeLPS(pat, M, lps);
        int i = 0;
        while (i < N) {
            if (pat.charAt(j) == txt.charAt(i)) { j++; i++; }
            if (j == M) return true; // Found
            else if (i < N && pat.charAt(j) != txt.charAt(i)) {
                if (j != 0) j = lps[j - 1]; else i = i + 1;
            }
        }
        return false;
    }
    static void computeLPS(String pat, int M, int[] lps) {
        int len = 0, i = 1; lps[0] = 0;
        while (i < M) {
            if (pat.charAt(i) == pat.charAt(len)) { len++; lps[i] = len; i++; }
            else { if (len != 0) len = lps[len - 1]; else { lps[i] = len; i++; } }
        }
    }

    // 97. Roman To Integer
    static int romanToInt(String s) {
        Map<Character, Integer> m = Map.of('I',1,'V',5,'X',10,'L',50,'C',100,'D',500,'M',1000);
        int res=0;
        for(int i=0; i<s.length(); i++) {
            if(i<s.length()-1 && m.get(s.charAt(i)) < m.get(s.charAt(i+1))) res -= m.get(s.charAt(i));
            else res += m.get(s.charAt(i));
        }
        return res;
    }

    // 99. Atoi (String to Int)
    static int myAtoi(String s) {
        s = s.trim();
        if(s.isEmpty()) return 0;
        int sign=1, i=0, r=0;
        if(s.charAt(0)=='-' || s.charAt(0)=='+') { sign = (s.charAt(0)=='-')? -1 : 1; i++; }
        while(i<s.length() && Character.isDigit(s.charAt(i))) {
            int d = s.charAt(i++) - '0';
            if(r > (Integer.MAX_VALUE - d)/10) return (sign==1)? Integer.MAX_VALUE : Integer.MIN_VALUE;
            r = r*10 + d;
        }
        return r * sign;
    }

    // 101. String Compression
    static String compress(String s) {
        StringBuilder sb = new StringBuilder();
        int c = 1;
        for(int i=0; i<s.length(); i++) {
            if(i<s.length()-1 && s.charAt(i)==s.charAt(i+1)) c++;
            else { sb.append(s.charAt(i)).append(c); c=1; }
        }
        return sb.toString();
    }
